Crie uma função que receba um array:
- Os valores devem ser separados por "," .
- Deve ser um array apenas de números.
- Retorne o menor número do array.

JavaScript:

                
                    function menorNum() {
                        let array = document.getElementById('array-c21').value;
                        result = array.split(',')
                        finalResult = []
                    
                        for(let i = 0; i ‹ result.length; i++) {
                            if(typeof parseFloat(result[i]) == 'number' && !isNaN(result[i])) {
                                finalResult.push(parseFloat(result[i]));
                            }
                        }
                    
                        const menor = Math.min(...finalResult);
